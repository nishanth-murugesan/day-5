Return all the palindromes in an array:
anonymous function:
var temp = [];
  var palindrome = function(arr){
      for(var i=0; i<arr.length; i++){
          var str = arr[i];
          var rev = arr[i].split("").reverse().join("");
          
          if(str == rev){
             temp.push(str);
          }
      }
    return temp;
  }
  console.log(palindrome(["121", "madam", "424", "guvi", "dad"]));
